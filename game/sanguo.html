// 内存溢出的话，多设置点内存

const notifier = require("node-notifier");

const fs = require("fs");

const UglifyJS = require("uglify-js");

const { parse } = require("babylon");

// javascript-obfuscator配置参数
const options = {

  // ]

// 最终合并的json键值对
var json = {};
// 开始索引值
var currentIndex = 1;
var index = 1;
var total = 0,
  init = true;
const rootPath = "./dev2",
  resultPath = "./dev3";
let round = 0;
readFiles(rootPath);
async function readFiles(path) {
  fs.readdir(path, function(err, data) {
    //获取总round
    if (init) {
      init = false;
      let repeat = false;
      for (let item of data) {
        const fileName = `${path}/${item}`;
        var stat = fs.lstatSync(fileName);
        if (stat.isDirectory()) {
          total++;
        } else if (!repeat) {
          repeat = true;
          total++;
        }
      }
    }
    console.log("total:", total);
    let flag = false;
    var newPath = "",
      rePath = "";
    if (err) {
      throw err;
    } else {
      console.log("文件名数组:", data);
      for (let item of data) {
        const fileName = `${path}/${item}`;
        console.log("文件名:", fileName);
        var stat = fs.lstatSync(fileName);
        newPath = fileName.replace(rootPath, resultPath);
        if (stat.isDirectory()) {
          console.log(`创建目录：${newPath}`);
          fs.mkdirSync(newPath);
          readFiles(fileName);
          continue;
        } else if (/.js$/.test(fileName)) {
          // 简单判断下如果是js文件
          rePath = path;
          mess(item, fileName, newPath);
          flag = true;
        }
      }
      if (flag) {
        flag = false;
        // 处理完全部的文件，把json输出到文件中
        const rePath2 = rePath.replace(rootPath, resultPath);
        console.log(`生成：${rePath2}/stringFile${round}.bin`);
        fs.writeFileSync(
          `${rePath2}/stringFile${round}.bin`,
          JSON.stringify(json)
        );
        //compressing.zip.compressFile(`${rePath2}/stringFile${round}.bin`, `${rePath2}/res${round}.zip`)

        round++;
        index = 1;
        currentIndex = 1;
        json = {};
      }

      if (round == total) {
        const title = "混淆结束";
        notifier.notify({
          title,
          message: `${round}/${total}`,
        });
      }
    }
  });
}

/**
 * mess用来对fileName源代码进行混淆、压缩、语法树生成替换字符串、并最终输出到newPath路径下
 * @param {*} item 要处理的文件名
 * @param {*} fileName 源文件路径
 * @param {*} newPath 输出文件路径
 */
async function mess(item, fileName, newPath) {
  console.log("mess:", item, fileName, newPath);
  // 读取源文件的内容
  const code = fs.readFileSync(fileName, "utf8");
  // 混淆 压缩 美化 -- start
  //console.log('开始混淆代码')
  var obfuscationResult = obfuscate(code, options);
  // 混淆后的代码
  //console.log('混淆结束，获取混淆后的代码，并压缩美化')
  var output = obfuscationResult.getObfuscatedCode();

  // 生成语法树
  var ast = parse(output);
  // 遍历语法树，并替换字符串
  //console.log('开始遍历语法树，并替换字符串')
  traverse(ast, {
    StringLiteral(path) {
      // 找到并遍历所有的字符串
      if (
        path.parent &&
        path.parent.type == "ObjectProperty" &&
        path.parent.key.type == "StringLiteral"
      ) {
        return;
      }
      index = currentIndex.toString(36); // 变为36进制
      json[`D${index}`] = path.node.value;
      path.replaceWith(
        t.expressionStatement(
          t.identifier(`WJSGDD${index}`)
          // t.memberExpression(
          //     t.memberExpression(
          //         t.identifier('wx'),
          //         t.identifier('$S')
          //     ),
          //     t.identifier(`D${index}`)
          // )
        )
      );
      currentIndex++;
      //console.log('i:', currentIndex)
    },
  });
  console.log("开始生成代码");
  output = generator(ast, {}, code).code;
  console.log("把代码写入目录:", newPath);
  fs.writeFileSync(newPath, UglifyJS.minify(output, {}).code, "utf8");
  console.log("写入成功");
}
