# 准备脚本

  1. 创建脚本文件夹
mkdir tool && cd tool
  1. 新建 package.json 文件，文件内容如下

  1. 新建 app.js 文件，文件内容如下
// 内存溢出的话，多设置点内存
// node --max-old-space-size=4096 app2.js

const fs = require("fs");
// 压缩混淆美化插件
const UglifyJS = require("uglify-js");
// 混淆插件
const { obfuscate } = require("javascript-obfuscator");
// ast相关插件
const { parse } = require("babylon");
const traverse = require("babel-traverse").default;
const t = require("babel-types");
const generator = require("babel-generator").default;
// 通知插件
const notifier = require("node-notifier");

// javascript-obfuscator配置参数
const options = {
  rotateUnicodeArray: true,
  compact: true,
  controlFlowFlattening: true,
  controlFlowFlatteningThreshold: 0.1,
  // controlFlowFlatteningThreshold: 0.75,
  deadCodeInjection: false,
  // deadCodeInjectionThreshold: 1,
  debugProtection: false,
  debugProtectionInterval: false,
  disableConsoleOutput: false,
  identifierNamesGenerator: "hexadecimal", // hexadecimal   mangled
  log: false,
  renameGlobals: false,
  rotateStringArray: false, // push【】shift【】
  // selfDefending: true,
  shuffleStringArray: true,
  splitStrings: true,
  splitStringsChunkLength: 10,
  stringArray: true,
  stringArrayEncoding: "rc4",
  // stringArrayThreshold: 0.75,
  stringArrayThreshold: 0.1,
  transformObjectKeys: false, // 王城争霸y读不到，把这个关掉，其他的开上吧
  unicodeEscapeSequence: false,
  // reservedStrings: [
  //     "wx\.[\$]O\.P[0-9a-z]{1,8}"
  // ]
};

// 最终合并的json键值对
var json = {};
// 开始索引值
var currentIndex = 1;
var index = 1;

fs.readdir("./dev2", function (err, data) {
  if (err) {
    throw err;
  } else {
    console.log("需要处理的文件：", data);
    data.map((item) => {
      // 需要处理的文件的数组，一个一个处理
      const fileName = `./dev2/${item}`;
      console.log("准备处理文件：", fileName);
      if (/.js$/.test(fileName)) {
        console.log(`${fileName}是js文件`);
        console.log("开始读取代码");
        const code = fs.readFileSync(fileName, "utf8");
        console.log("开始混淆代码");
        var obfuscationResult = obfuscate(code, options);
        // 混淆后的代码
        console.log("混淆结束，获取混淆后的代码");
        var output = obfuscationResult.getObfuscatedCode();
        if (true) {
          // ------------使用ast的方式抽取字符串--------------------
          // 生成语法树
          console.log("生成语法树");
          var ast = parse(output);
          // 遍历语法树
          console.log("遍历语法树");
          traverse(ast, {
            StringLiteral(path) {
              // 找到并遍历所有的字符串
              if (
                path.parent &&
                path.parent.type == "ObjectProperty" &&
                path.parent.key.type == "StringLiteral"
              ) {
                return;
              }

              index = currentIndex.toString(36); // 变为36进制
              // 把匹配到的字符串保存到json中，
              json[`X${index}`] = path.node.value;
              path.replaceWith(
                t.expressionStatement(t.identifier(`_x.X${index}`))
              );
              currentIndex++;
              console.log("i:", currentIndex);
            },
          });
          console.log("开始生成代码");
          output = generator(ast, {}, code).code;
          output = "var _x=wx.$X;" + output;
        }
        console.log("把代码写入dev3目录下");
        fs.writeFileSync(
          `./dev3/${item}`,
          UglifyJS.minify(output, {}).code,
          "utf8"
        );
        console.log("写入成功");
      }
    });
    // 处理完全部的文件，把json输出到文件中
    fs.writeFileSync(`./dev3/string.bin`, JSON.stringify(json));
    // 压缩
    var AdmZip = require("adm-zip");
    var zip = new AdmZip();
    zip.addLocalFile("./dev3/string.bin");
    zip.writeZip("./dev3/string.zip");
    fs.unlinkSync("./dev3/string.bin");
    notifier.notify("混淆完成");
  }
});
  1. 安装 npm 包
npm i
  1. 创建 dev1 和 dev3 文件夹
mkdir dev1
mkdir dev3

# 需要混淆的文件

  • js目录下除了egret.min.js和jszip.min.js这2个文件其他全6个全混
  • library目录下的5个文件全混
  • stage1/default.min.js
  • stage2/main.min.js

共6+5+1+1（13）个文件

# 执行混淆操作

  • 把需混淆的文件放入脚本项目文件夹下的 dev1 目录下，并执行如下命令，等待输出结果即可
npm run app
  • 结果会输出到 dev3 目录下
