# 默认权限

  • 新建普通文件默认权限为666-umask
  • 新建文件夹默认权限为777-umask
umask # 查看umask属性
umask 002 # 设置umask属性
umask 0002 # 设置umask属性

# 特殊权限

  • SUID
  • SGID
  • SBIT（Sticky BIT）
# 设置权限：
chmod o+t /tmp/common
# 取消权限：
chmod o-t /tmp/common
# 设置权限：
chmod 1755 /tmp/common
# 取消：
chmod 0755 /tmp/common
