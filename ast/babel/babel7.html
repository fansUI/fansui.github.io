# 1. 创建新项目文件夹

// 创建文件夹
mkdir babel7-demo
// 进入babel7-demo文件夹
cd babel7-demo
// 初始化项目
npm init -y

# 2. 安装@babel/cli命令行工具

npm i -D @babel/cli

# 3. 创建src目录并新建src/index.js文件

mkdir src
touch src/index.js

# 4. index.js内容如下

const a = 123;

const add = (a, b) => a + b;

# 5. babel 转换

babel src/index.js

执行结果报错如下：

// 命令没有发现：babel
command not found: babel

是因为 @babel/cli 命令行工具我们是局部安装的，需要使用下述命令执行

npx babel src/index.js

我们发现 babel 并没有生效，代码还是原样输出，并没有进行转换?
是因为 babel 不知道该如何转换，我们需要给 babel 制定规则

# 6. 制定（指定）规则：插件和预设

npm i -D @babel/preset-env
  • 方式一：在项目根目录创建.babelrc或.babelrc.json或.babelrc.js或.babelrc.cjs文件，文件内容如下
// .babelrc 或 .babelrc.json
{
    "presets": ["@babel/preset-env"],
    "plugins": []
}

// .babelrc.js 或 .babelrc.cjs
module.exports = {
  presets: ["@babel/preset-env"],
  plugins: [],
};
  • 方式二：在项目根目录创建babel.config.js或.babelrc.cjs文件，文件内容如下
module.exports = {
  presets: ["@babel/preset-env"],
  plugins: [],
};

// 函数形式，需要参数并调用cache方法
module.exports =  (api)=>{
    api.cache(true);
    return {
        "presets": ["@babel/env"]
    }
}
  • 方式三：在项目根目录创建babel.config.json文件，文件内容如下
{
  presets: ["@babel/preset-env"],
  plugins: [],
};
  • 方式四：在package.json文件中添加babel字段
"babel": {
    "presets": ["@babel/preset-env"]
}

执行如下命令：可以看到转换的结果

npx babel src/index.js

// 转换后
"use strict";

var a = 123;

var add = function add(a, b) {
  return a + b;
};

这时候转换后的代码是打印在屏幕上的，我们需要把它输出到文件中，如下

// 把转换后的代码写入到src目录下的res.js文件中，
// -o：指定输出文件路径和名称，-o是--out-file的简写形式
npx babel src/index.js -o src/res.js
npx babel src/index.js --out-file src/res.js

# 7. 针对文件夹进行转换

// 把src目录下的js文件使用babel转换并输出到dist目录下
npx babel src -d dist
npx babel src --out-dir dist

# 8. 监听文件改动，自动转换

// 针对文件
npx babel src/index.js -w -o src/res.js
npx babel src/index.js --watch -o src/res.js

// 针对文件夹
npx babel src -w -d dist
npx babel src --watch -d dist

# 9. 使用 npm 脚本

// package.json

"scripts": {
    "build": "babel src -d dist"
}
