# shell脚本

  • 脚本声明

  • 脚本注释

  • 脚本功能

  • shell脚本文件以sh为扩展符

  • shell脚本第一行为声明：如下


#!/bin/bash

# this is a shell file

cd ~

ls -la

# shell脚本内置变量，从参数传递
echo $0   # $0：对应脚本文件名称
echo $1   # $1：对应第一个参数
echo $2   # $2：对应第二个参数...以此类推
echo $#   # $#：对应参数的个数
echo $*   # $*：对应全部的参数
echo $?   # $?：对应上一次的命令是否成功，0：成功，其他：不成功

# 接收参数
# 处理参数



# 测试文件类型、逻辑、测试（比较）数字大小、字符串


# 条件测试语法
# shell脚本的测试语句：[ 测试的内容 ]

# 判断路径是否是目录
[ -d /etc ] # 中括号的内容一中括号之间何有空格，中括号是格式：用来测试
echo $? # 上一个命令是否执行成功，0：成功，其他：不成功


# 判断路径是否是文件
[ -f /etc/passwd ] # 中括号的内容一中括号之间何有空格，中括号是格式：用来测试
echo $? # 上一个命令是否执行成功，0：成功，其他：不成功

# 判断路径是否存在
[ -e /etc ]
[ -e /etc/passwd ]

echo $USER # 当前登录用户

# 判断$USER是否等于root
[ $USER = root ]

# -eq
# -nq
# -lt
# -le
# -gt
# -ge

# 判断空闲内存是否大于1024M
free # 查看内存信息，以K为单位
free -h # 查看内存信息，以G为单位
free -m # 查看内存信息，以M为单位
free -m | grep Mem: # 筛选出内存信息那一行
free -m | grep Mem: | awk '{print $4}' # 筛选出内存信息那一行的第四列
`free -m | grep Mem: | awk '{print $4}'` # 使用反引号包裹
[ `free -m | grep Mem: | awk '{print $4}'` -lt 1024 ] && echo '小于1024' || echo '大于1024'


# 判断变量是否定义过（是否使用过）
[ -z $USER ]
echo $?


# 判断用户输入是否大于等于90
read -p "Enter a number:" INPUT
if [ $INPUT -ge 90 ]
then
echo '您输入的数字大于等于90'
fi

# 判断用户输入是否大于等于90
read -p "Enter a number:" INPUT
if [ $INPUT -ge 90 ]
then
echo '您输入的数字大于等于90'
else
echo '您输入的数字小于90'
fi
