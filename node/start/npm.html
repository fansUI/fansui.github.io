# 查看 npm 所有命令

npm;

# 查看单个命令的用法

npm help install
npm help login

# 初始化项目

npm init
// 或
npm init -y

# 安装软件包（项目依赖）

npm i express
npm i express -S
npm i express --save
npm install express
npm install express -S
npm install express --save

# 安装软件包（开发依赖）

npm i express -D
npm i express --save-dev
npm install express -D
npm install express --save-dev

# 移除软件包（项目依赖）

npm un express
npm un express -S
npm un express --save
npm uninstall express
npm uninstall express -S
npm uninstall express --save

# 移除软件包（开发依赖）

npm un express -D
npm uninstall express -D
npm un express --save-dev
npm uninstall express --save-dev

# 更新软件包

// 更新全部软件包
npm update
// 更新单个软件包
npm update express

# 查看全局安装路径

npm root -g

# 查看当前项目安装路径

npm root

# 查看安装的软件包

npm list
// 查看全局安装的软件包
npm list -g
npm list -g --depth 0

# 查看软件包最新的可用版本

// npm view [包名] version
npm view vue version

# 查看软件包所有版本

npm view vue versions

# 查看软件包详细信息

// npm view [包名]
npm view vue

# npm 脚本

npm run dev
npm start
npm run build
npm run test

# npm 登录

npm login

# 查看当前登录的用户

npm whoami

# 退出当前用户

npm logout

# 发布

npm publish

# 撤销发布

npm unpublish
